---
title: 06.根据不同请求响应不同的内容
tags: NodeJS
categories: NodeJS
date: 2017-11-1
---

~~~javascript
//引入http核心模块
var http = require("http");

//创建一个服务器对象
var server = http.createServer();

//开启服务器
server.on("request",function(req,res){
    //1.0接收请求的地址
    var url = req.url;
    //2.0根据不同的请求响应不同的内容
    if(url == "/index") {
        res.write("index");
    } else if(url == "/abc") {
        res.write("abc");
    } else {
        res.write("i don't know");
    }
    res.end();
});

//开启监听
server.listen("3000",function(){
   console.log("running"); 
});
~~~
