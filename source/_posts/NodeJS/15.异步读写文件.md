title: 15.异步读写文件
tags: NodeJS
categories: NodeJS
date: 2017-11-1
---

```javascript
var fs = require("fs");

console.log("a文件开始读取");
//读取a文件
fs.readFile("./data/a.txt",function(err,data){
    if(err) {
        return console.log("a失败");
    }
    console.log("a文件读取完成");
    console.log(data.toString());
});

console.log("b文件开始读取");
//读取b文件
fs.readFile("./data",function(err,data){
    if(err) {
        return console.log("b失败");
    }
    console.log("b文件读取完成");
    console.log(data.toString());
});

console.log("c文件开始读取");
//读取c文件
fs.readFile("./data/c.txt",function(err,data){
    if(err) {
        return console.log("c失败");
    }
    console.log("c文件读取完成");
    console.log(data.toString());
});
```
