```javascript
//引入express文件
const express = require("express");

//创建服务器
var app = express();

//配置路由：
//得到主页：/  get         --->index.html
app.get("/",function(req,res){
    res.send("您请求的是index.html");
});

//得到新增页面：/add   get     ---->add.html
app.get("/add",function(req,res){
    res.send("您请求的是add.html");
});

//提交新增数据：/add   post    ---->处理新增提交数据的逻辑
app.post("/add",function(req,res){
    res.send("处理新增的提交数据中");
});


//开启监听
app.listen(3000,function(){
    console.log("running");
});
```
