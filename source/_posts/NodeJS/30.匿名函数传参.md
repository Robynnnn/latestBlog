```javascript
function fn(callback) {
    var a = 1;
    var b = 2;
    var c = a + b;
    callback(a,b,c);
}
// fn(function(num1,num2,num3){
//     console.log(num1 + num2 + num3);
// });
var demo = function(num1,num2,num3){
    console.log(num1+ num2+num3);
}
fn(demo);
```
