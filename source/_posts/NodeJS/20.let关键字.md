```javascript
//在ES6中，如果要声明变量除了可以用var关键字，还可以使用let关键字：
// var a = 1234;
// console.log(a);
// let a = 123;
// console.log(a);

//区别：
//1)let声明的变量没有变量提升
// console.log(a);//undefined
// var a = 123;
// console.log(a);//123
// console.log(a);//报错：a is not defined
// let a = 123;
// console.log(a);//123

//2)let声明的变量有块级作用域
//作用域：后台语言的概念：java php C#
//  凡是大括号包含的区域都叫块级作用,块级作用域中的变量会在这个作用域中的代码执行完之后全部销毁。
// if(true) {
//     var a = 123;
//     console.log(a);
// }
// console.log(a);
// if(true) {
//     let a = 123;
//     console.log(a);
// }
// console.log(a);

//3) let 申明的全局变量在函数内部也可以使用(区别PHP)
// let a = 1; 
// function fn() {
//     console.log(a);
// }
```
